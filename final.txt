## Exercise 5.1: Basic Gossip Protocol

### Goal
Implement a gossip protocol where nodes forward messages probabilistically.

### Code: `scratch/gossip-protocol.cc`
```cpp
#include "ns3/core-module.h"
#include "ns3/network-module.h"
#include "ns3/internet-module.h"
#include "ns3/wifi-module.h"
#include "ns3/mobility-module.h"

using namespace ns3;

NS_LOG_COMPONENT_DEFINE("GossipProtocol");

struct GossipMessage
{
    uint32_t messageId;
    uint32_t originNode;
    uint32_t hopCount;
    double timestamp;
    char payload[64];
};

class GossipApp : public Application
{
public:
    GossipApp();
    virtual ~GossipApp();
    
    void Setup(uint16_t port, double gossipProb, uint32_t maxHops);
    void InitiateGossip(std::string message);

private:
    virtual void StartApplication(void);
    virtual void StopApplication(void);
    
    void HandleRead(Ptr<Socket> socket);
    void ForwardGossip(GossipMessage msg);
    bool ShouldForward(uint32_t messageId);
    
    Ptr<Socket> m_socket;
    uint16_t m_port;
    double m_gossipProb;
    uint32_t m_maxHops;
    std::set<uint32_t> m_seenMessages;
    uint32_t m_messageCounter;
};

GossipApp::GossipApp()
    : m_socket(0),
      m_port(0),
      m_gossipProb(0.8),
      m_maxHops(10),
      m_messageCounter(0)
{
}

GossipApp::~GossipApp()
{
    m_socket = 0;
}

void GossipApp::Setup(uint16_t port, double gossipProb, uint32_t maxHops)
{
    m_port = port;
    m_gossipProb = gossipProb;
    m_maxHops = maxHops;
}

void GossipApp::StartApplication(void)
{
    m_socket = Socket::CreateSocket(GetNode(), UdpSocketFactory::GetTypeId());
    m_socket->SetAllowBroadcast(true);
    
    InetSocketAddress local = InetSocketAddress(Ipv4Address::GetAny(), m_port);
    m_socket->Bind(local);
    m_socket->SetRecvCallback(MakeCallback(&GossipApp::HandleRead, this));
}

void GossipApp::StopApplication(void)
{
    if (m_socket)
    {
        m_socket->Close();
    }
    
    NS_LOG_INFO("Node " << GetNode()->GetId() << " saw " 
                << m_seenMessages.size() << " unique messages");
}

void GossipApp::InitiateGossip(std::string message)
{
    GossipMessage msg;
    msg.messageId = m_messageCounter++;
    msg.originNode = GetNode()->GetId();
    msg.hopCount = 0;
    msg.timestamp = Simulator::Now().GetSeconds();
    strncpy(msg.payload, message.c_str(), sizeof(msg.payload) - 1);
    msg.payload[sizeof(msg.payload) - 1] = '\0';
    
    m_seenMessages.insert(msg.messageId);
    
    NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Node " 
                << GetNode()->GetId() << " initiated gossip: \"" 
                << msg.payload << "\" (ID: " << msg.messageId << ")");
    
    ForwardGossip(msg);
}

bool GossipApp::ShouldForward(uint32_t messageId)
{
    // Check if already seen
    if (m_seenMessages.find(messageId) != m_seenMessages.end())
    {
        return false;
    }
    
    // Probabilistic forwarding
    Ptr<UniformRandomVariable> rand = CreateObject<UniformRandomVariable>();
    return rand->GetValue() < m_gossipProb;
}

void GossipApp::ForwardGossip(GossipMessage msg)
{
    msg.hopCount++;
    
    if (msg.hopCount > m_maxHops)
    {
        NS_LOG_INFO("Node " << GetNode()->GetId() 
                    << " dropping message " << msg.messageId 
                    << " (max hops reached)");
        return;
    }
    
    Ptr<Packet> packet = Create<Packet>((uint8_t*)&msg, sizeof(msg));
    
    InetSocketAddress remote = InetSocketAddress(Ipv4Address("255.255.255.255"), m_port);
    m_socket->SendTo(packet, 0, remote);
    
    NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Node " 
                << GetNode()->GetId() << " forwarding message " 
                << msg.messageId << " (hop " << msg.hopCount << ")");
}

void GossipApp::HandleRead(Ptr<Socket> socket)
{
    Ptr<Packet> packet;
    Address from;
    
    while ((packet = socket->RecvFrom(from)))
    {
        if (packet->GetSize() >= sizeof(GossipMessage))
        {
            GossipMessage msg;
            packet->CopyData((uint8_t*)&msg, sizeof(msg));
            
            // Don't process if already seen
            if (m_seenMessages.find(msg.messageId) != m_seenMessages.end())
            {
                continue;
            }
            
            m_seenMessages.insert(msg.messageId);
            
            NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Node " 
                        << GetNode()->GetId() << " received gossip message " 
                        << msg.messageId << " from origin " << msg.originNode 
                        << " (hop " << msg.hopCount << "): \"" << msg.payload << "\"");
            
            // Decide whether to forward
            if (ShouldForward(msg.messageId))
            {
                // Small random delay before forwarding
                Ptr<UniformRandomVariable> delay = CreateObject<UniformRandomVariable>();
                double delayTime = delay->GetValue(0.01, 0.1);
                
                Simulator::Schedule(Seconds(delayTime), &GossipApp::ForwardGossip, this, msg);
            }
        }
    }
}

int main(int argc, char *argv[])
{
    LogComponentEnable("GossipProtocol", LOG_LEVEL_INFO);
    
    uint32_t nNodes = 10;
    double gossipProb = 0.8;
    uint32_t maxHops = 5;
    
    // Create nodes
    NodeContainer nodes;
    nodes.Create(nNodes);
    
    // WiFi setup
    YansWifiChannelHelper channel = YansWifiChannelHelper::Default();
    YansWifiPhyHelper phy;
    phy.SetChannel(channel.Create());
    
    WifiHelper wifi;
    wifi.SetRemoteStationManager("ns3::AarfWifiManager");
    
    WifiMacHelper mac;
    mac.SetType("ns3::AdhocWifiMac");
    
    NetDeviceContainer devices = wifi.Install(phy, mac, nodes);
    
    // Random positions
    MobilityHelper mobility;
    mobility.SetPositionAllocator("ns3::RandomRectanglePositionAllocator",
                                  "X", StringValue("ns3::UniformRandomVariable[Min=0.0|Max=100.0]"),
                                  "Y", StringValue("ns3::UniformRandomVariable[Min=0.0|Max=100.0]"));
    mobility.SetMobilityModel("ns3::ConstantPositionMobilityModel");
    mobility.Install(nodes);
    
    // Internet stack
    InternetStackHelper stack;
    stack.Install(nodes);
    
    Ipv4AddressHelper address;
    address.SetBase("10.1.1.0", "255.255.255.0");
    Ipv4InterfaceContainer interfaces = address.Assign(devices);
    
    // Install gossip apps
    for (uint32_t i = 0; i < nodes.GetN(); i++)
    {
        Ptr<GossipApp> app = CreateObject<GossipApp>();
        app->Setup(9, gossipProb, maxHops);
        nodes.Get(i)->AddApplication(app);
        app->SetStartTime(Seconds(0.0));
        app->SetStopTime(Seconds(30.0));
    }
    
    // Node 0 initiates gossip at t=1s
    Ptr<GossipApp> app0 = DynamicCast<GossipApp>(nodes.Get(0)->GetApplication(0));
    Simulator::Schedule(Seconds(1.0), &GossipApp::InitiateGossip, app0, 
                       "Hello from Node 0!");
    
    // Node 5 initiates different gossip at t=5s
    Ptr<GossipApp> app5 = DynamicCast<GossipApp>(nodes.Get(5)->GetApplication(0));
    Simulator::Schedule(Seconds(5.0), &GossipApp::InitiateGossip, app5, 
                       "News from Node 5!");
    
    Simulator::Stop(Seconds(30.0));
    Simulator::Run();
    Simulator::Destroy();
    return 0;
}
```

### Build and Run
```bash

```

### Key Concepts

#### Gossip Protocol Features
1. **Message Deduplication**: Track seen messages
2. **Probabilistic Forwarding**: Forward with probability p
3. **Hop Limit**: Prevent infinite propagation
4. **Random Delays**: Avoid network congestion

#### Gossip Parameters
```cpp
gossipProb = 0.8;   // 80% chance of forwarding
maxHops = 5;        // Maximum 5 hops
```

### Goal

```cpp
#include "ns3/core-module.h"

using namespace ns3;

NS_LOG_COMPONENT_DEFINE("InterferenceZones");

class InterferenceZone
{
public:
    InterferenceZone(Vector center, double radius, double attenuation)
        : m_center(center), m_radius(radius), m_attenuation(attenuation)
    {
    }
    
    bool IsInZone(Vector position) const
    {
        double dx = position.x - m_center.x;
        double dy = position.y - m_center.y;
        double distance = std::sqrt(dx * dx + dy * dy);
        return distance <= m_radius;
    }
    
    double GetAttenuation() const { return m_attenuation; }
    Vector GetCenter() const { return m_center; }
    double GetRadius() const { return m_radius; }

private:
    Vector m_center;
    double m_radius;
    double m_attenuation;  // Additional loss in dB
};

class InterferenceModel : public PropagationLossModel
{
public:
    static TypeId GetTypeId(void);
    InterferenceModel();
    
    void AddInterferenceZone(Vector center, double radius, double attenuation);
    
private:
    virtual double DoCalcRxPower(double txPowerDbm, 
                                Ptr<MobilityModel> a, 
                                Ptr<MobilityModel> b) const;
    virtual int64_t DoAssignStreams(int64_t stream);
    
    std::vector<InterferenceZone> m_zones;
    Ptr<PropagationLossModel> m_baseModel;
};

TypeId InterferenceModel::GetTypeId(void)
{
    static TypeId tid = TypeId("InterferenceModel")
        .SetParent<PropagationLossModel>()
        .SetGroupName("Propagation")
        .AddConstructor<InterferenceModel>();
    return tid;
}

InterferenceModel::InterferenceModel()
{
    m_baseModel = CreateObject<LogDistancePropagationLossModel>();
}

void InterferenceModel::AddInterferenceZone(Vector center, double radius, double attenuation)
{
    m_zones.push_back(InterferenceZone(center, radius, attenuation));
    NS_LOG_INFO("Added interference zone at (" << center.x << ", " << center.y 
                << ") with radius " << radius << "m and " 
                << attenuation << "dB attenuation");
}

double InterferenceModel::DoCalcRxPower(double txPowerDbm,
                                       Ptr<MobilityModel> a,
                                       Ptr<MobilityModel> b) const
{
    // Start with base propagation loss
    double rxPowerDbm = m_baseModel->CalcRxPower(txPowerDbm, a, b);
    
    Vector posA = a->GetPosition();
    Vector posB = b->GetPosition();
    
    // Check if either node is in an interference zone
    for (const auto& zone : m_zones)
    {
        if (zone.IsInZone(posA) || zone.IsInZone(posB))
        {
            rxPowerDbm -= zone.GetAttenuation();
            
            NS_LOG_DEBUG("Interference detected: " << zone.GetAttenuation() 
                        << "dB additional loss");
        }
    }
    
    return rxPowerDbm;
}
```

## Exercise 5.3: Final Project - Complete System

### Goal
Integrate everything: mobile nodes with waypoints, gossip protocol, broadcasting, interference zones.

### Code: `scratch/final-project.cc`
```cpp
#include "ns3/core-module.h"
#include "ns3/network-module.h"
#include "ns3/internet-module.h"
#include "ns3/wifi-module.h"
#include "ns3/mobility-module.h"
#include "ns3/netanim-module.h"

using namespace ns3;

NS_LOG_COMPONENT_DEFINE("FinalProject");

// Message types
enum MessageType
{
    BEACON = 1,           // Mobile node beacon
    RESPONSE = 2,         // Static node response
    GOSSIP = 3           // Gossip message
};

struct MessageHeader
{
    uint8_t type;
    uint32_t nodeId;
    uint32_t messageId;
    double x, y;
    double timestamp;
    uint32_t hopCount;
    char payload[128];
};

class CompleteSystemApp : public Application
{
public:
    CompleteSystemApp();
    virtual ~CompleteSystemApp();
    
    void Setup(uint16_t port, bool isMobile, double beaconInterval, 
               double gossipProb, uint32_t maxHops);
    void InitiateGossip(std::string message);

private:
    virtual void StartApplication(void);
    virtual void StopApplication(void);
    
    void SendBeacon(void);
    void SendResponse(uint32_t targetNode, Address targetAddress);
    void ForwardGossip(MessageHeader msg);
    void HandleRead(Ptr<Socket> socket);
    bool ShouldForwardGossip(uint32_t messageId);
    
    Ptr<Socket> m_socket;
    uint16_t m_port;
    bool m_isMobile;
    double m_beaconInterval;
    double m_gossipProb;
    uint32_t m_maxHops;
    EventId m_beaconEvent;
    std::set<uint32_t> m_seenMessages;
    uint32_t m_messageCounter;
    std::map<uint32_t, uint32_t> m_detectionCount;
};

CompleteSystemApp::CompleteSystemApp()
    : m_socket(0),
      m_port(0),
      m_isMobile(false),
      m_beaconInterval(2.0),
      m_gossipProb(0.8),
      m_maxHops(10),
      m_messageCounter(0)
{
}

CompleteSystemApp::~CompleteSystemApp()
{
    m_socket = 0;
}

void CompleteSystemApp::Setup(uint16_t port, bool isMobile, double beaconInterval,
                              double gossipProb, uint32_t maxHops)
{
    m_port = port;
    m_isMobile = isMobile;
    m_beaconInterval = beaconInterval;
    m_gossipProb = gossipProb;
    m_maxHops = maxHops;
}

void CompleteSystemApp::StartApplication(void)
{
    m_socket = Socket::CreateSocket(GetNode(), UdpSocketFactory::GetTypeId());
    m_socket->SetAllowBroadcast(true);
    
    InetSocketAddress local = InetSocketAddress(Ipv4Address::GetAny(), m_port);
    m_socket->Bind(local);
    m_socket->SetRecvCallback(MakeCallback(&CompleteSystemApp::HandleRead, this));
    
    if (m_isMobile)
    {
        SendBeacon();
    }
}

void CompleteSystemApp::StopApplication(void)
{
    if (m_beaconEvent.IsRunning())
    {
        Simulator::Cancel(m_beaconEvent);
    }
    
    if (m_socket)
    {
        m_socket->Close();
    }
    
    // Print statistics
    if (m_isMobile)
    {
        std::cout << "\nMobile Node " << GetNode()->GetId() << " Statistics:" << std::endl;
        std::cout << "  Detected by:" << std::endl;
        for (auto& pair : m_detectionCount)
        {
            std::cout << "    Node " << pair.first << ": " << pair.second << " times" << std::endl;
        }
        std::cout << "  Saw " << m_seenMessages.size() << " unique gossip messages" << std::endl;
    }
}

void CompleteSystemApp::SendBeacon(void)
{
    Ptr<MobilityModel> mobility = GetNode()->GetObject<MobilityModel>();
    Vector pos = mobility->GetPosition();
    
    MessageHeader header;
    header.type = BEACON;
    header.nodeId = GetNode()->GetId();
    header.messageId = 0;
    header.x = pos.x;
    header.y = pos.y;
    header.timestamp = Simulator::Now().GetSeconds();
    header.hopCount = 0;
    strcpy(header.payload, "BEACON");
    
    Ptr<Packet> packet = Create<Packet>((uint8_t*)&header, sizeof(header));
    
    InetSocketAddress remote = InetSocketAddress(Ipv4Address("255.255.255.255"), m_port);
    m_socket->SendTo(packet, 0, remote);
    
    NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Mobile Node " 
                << GetNode()->GetId() << " sent beacon from (" 
                << pos.x << ", " << pos.y << ")");
    
    m_beaconEvent = Simulator::Schedule(Seconds(m_beaconInterval), 
                                       &CompleteSystemApp::SendBeacon, this);
}

void CompleteSystemApp::SendResponse(uint32_t targetNode, Address targetAddress)
{
    Ptr<MobilityModel> mobility = GetNode()->GetObject<MobilityModel>();
    Vector pos = mobility->GetPosition();
    
    MessageHeader header;
    header.type = RESPONSE;
    header.nodeId = GetNode()->GetId();
    header.messageId = 0;
    header.x = pos.x;
    header.y = pos.y;
    header.timestamp = Simulator::Now().GetSeconds();
    header.hopCount = 0;
    strcpy(header.payload, "RESPONSE");
    
    Ptr<Packet> packet = Create<Packet>((uint8_t*)&header, sizeof(header));
    m_socket->SendTo(packet, 0, targetAddress);
    
    NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Static Node " 
                << GetNode()->GetId() << " sent response to Node " << targetNode);
}

void CompleteSystemApp::InitiateGossip(std::string message)
{
    MessageHeader header;
    header.type = GOSSIP;
    header.nodeId = GetNode()->GetId();
    header.messageId = m_messageCounter++;
    
    Ptr<MobilityModel> mobility = GetNode()->GetObject<MobilityModel>();
    Vector pos = mobility->GetPosition();
    header.x = pos.x;
    header.y = pos.y;
    header.timestamp = Simulator::Now().GetSeconds();
    header.hopCount = 0;
    strncpy(header.payload, message.c_str(), sizeof(header.payload) - 1);
    header.payload[sizeof(header.payload) - 1] = '\0';
    
    uint32_t fullId = (header.nodeId << 16) | header.messageId;
    m_seenMessages.insert(fullId);
    
    NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Node " 
                << GetNode()->GetId() << " initiated gossip: \"" 
                << header.payload << "\"");
    
    ForwardGossip(header);
}

bool CompleteSystemApp::ShouldForwardGossip(uint32_t messageId)
{
    if (m_seenMessages.find(messageId) != m_seenMessages.end())
    {
        return false;
    }
    
    Ptr<UniformRandomVariable> rand = CreateObject<UniformRandomVariable>();
    return rand->GetValue() < m_gossipProb;
}

void CompleteSystemApp::ForwardGossip(MessageHeader header)
{
    header.hopCount++;
    
    if (header.hopCount > m_maxHops)
    {
        return;
    }
    
    Ptr<Packet> packet = Create<Packet>((uint8_t*)&header, sizeof(header));
    
    InetSocketAddress remote = InetSocketAddress(Ipv4Address("255.255.255.255"), m_port);
    m_socket->SendTo(packet, 0, remote);
    
    NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Node " 
                << GetNode()->GetId() << " forwarded gossip (hop " 
                << header.hopCount << ")");
}

void CompleteSystemApp::HandleRead(Ptr<Socket> socket)
{
    Ptr<Packet> packet;
    Address from;
    
    while ((packet = socket->RecvFrom(from)))
    {
        if (packet->GetSize() >= sizeof(MessageHeader))
        {
            MessageHeader header;
            packet->CopyData((uint8_t*)&header, sizeof(header));
            
            if (header.nodeId == GetNode()->GetId())
                continue;
            
            Ptr<MobilityModel> mobility = GetNode()->GetObject<MobilityModel>();
            Vector myPos = mobility->GetPosition();
            
            if (header.type == BEACON && !m_isMobile)
            {
                // Static node detected mobile node beacon
                double distance = std::sqrt(std::pow(myPos.x - header.x, 2) + 
                                          std::pow(myPos.y - header.y, 2));
                
                NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Static Node " 
                            << GetNode()->GetId() << " detected Mobile Node " 
                            << header.nodeId << " at distance " << distance << "m");
                
                SendResponse(header.nodeId, from);
            }
            else if (header.type == RESPONSE && m_isMobile)
            {
                // Mobile node received response
                m_detectionCount[header.nodeId]++;
                
                NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Mobile Node " 
                            << GetNode()->GetId() << " received response from Static Node " 
                            << header.nodeId);
            }
            else if (header.type == GOSSIP)
            {
                // Handle gossip message
                uint32_t fullId = (header.nodeId << 16) | header.messageId;
                
                if (m_seenMessages.find(fullId) != m_seenMessages.end())
                {
                    continue;
                }
                
                m_seenMessages.insert(fullId);
                
                NS_LOG_INFO("[" << Simulator::Now().GetSeconds() << "s] Node " 
                            << GetNode()->GetId() << " received gossip from Node " 
                            << header.nodeId << " (hop " << header.hopCount 
                            << "): \"" << header.payload << "\"");
                
                if (ShouldForwardGossip(fullId))
                {
                    Ptr<UniformRandomVariable> delay = CreateObject<UniformRandomVariable>();
                    double delayTime = delay->GetValue(0.01, 0.1);
                    
                    Simulator::Schedule(Seconds(delayTime), 
                                      &CompleteSystemApp::ForwardGossip, this, header);
                }
            }
        }
    }
}

int main(int argc, char *argv[])
{
    LogComponentEnable("FinalProject", LOG_LEVEL_INFO);
    
    // Parameters
    uint32_t nStaticNodes = 5;
    uint32_t nMobileNodes = 3;
    double simTime = 60.0;
    
    // Create nodes
    NodeContainer staticNodes;
    staticNodes.Create(nStaticNodes);
    
    NodeContainer mobileNodes;
    mobileNodes.Create(nMobileNodes);
    
    NodeContainer allNodes;
    allNodes.Add(staticNodes);
    allNodes.Add(mobileNodes);
    
    // WiFi setup
    YansWifiChannelHelper channel = YansWifiChannelHelper::Default();
    YansWifiPhyHelper phy;
    phy.SetChannel(channel.Create());
    
    WifiHelper wifi;
    wifi.SetRemoteStationManager("ns3::AarfWifiManager");
    
    WifiMacHelper mac;
    mac.SetType("ns3::AdhocWifiMac");
    
    NetDeviceContainer devices = wifi.Install(phy, mac, allNodes);
    
    // Static nodes in a pentagon
    MobilityHelper staticMobility;
    Ptr<ListPositionAllocator> staticPos = CreateObject<ListPositionAllocator>();
    double radius = 50.0;
    for (uint32_t i = 0; i < nStaticNodes; i++)
    {
        double angle = 2.0 * M_PI * i / nStaticNodes;
        double x = radius * std::cos(angle);
        double y = radius * std::sin(angle);
        staticPos->Add(Vector(x, y, 0.0));
    }
    
    staticMobility.SetPositionAllocator(staticPos);
    staticMobility.SetMobilityModel("ns3::ConstantPositionMobilityModel");
    staticMobility.Install(staticNodes);
    
    // Mobile nodes with waypoint paths
    for (uint32_t i = 0; i < nMobileNodes; i++)
    {
        Ptr<WaypointMobilityModel> waypoint = CreateObject<WaypointMobilityModel>();
        mobileNodes.Get(i)->AggregateObject(waypoint);
        
        // Create custom path for each mobile node
        double startX = -60.0 + i * 30.0;
        double startY = -60.0;
        waypoint->SetPosition(Vector(startX, startY, 0.0));
        
        // Add waypoints (create a path through the network)
        double time = 1.0;
        double speed = 5.0;
        
        std::vector<Vector> path = {
            Vector(startX, 0.0, 0.0),
            Vector(0.0, startY + 60.0, 0.0),
            Vector(-startX, 0.0, 0.0),
            Vector(0.0, -startY - 60.0, 0.0),
            Vector(startX, startY, 0.0)
        };
        
        Vector currentPos = Vector(startX, startY, 0.0);
        for (const auto& waypoint_pos : path)
        {
            double distance = CalculateDistance(currentPos, waypoint_pos);
            time += distance / speed;
            waypoint->AddWaypoint(Waypoint(Seconds(time), waypoint_pos));
            currentPos = waypoint_pos;
        }
    }
    
    // Internet stack
    InternetStackHelper stack;
    stack.Install(allNodes);
    
    Ipv4AddressHelper address;
    address.SetBase("10.1.1.0", "255.255.255.0");
    Ipv4InterfaceContainer interfaces = address.Assign(devices);
    
    // Install applications on static nodes
    for (uint32_t i = 0; i < staticNodes.GetN(); i++)
    {
        Ptr<CompleteSystemApp> app = CreateObject<CompleteSystemApp>();
        app->Setup(9, false, 0, 0.7, 5);  // static, no beacons, gossip enabled
        staticNodes.Get(i)->AddApplication(app);
        app->SetStartTime(Seconds(0.0));
        app->SetStopTime(Seconds(simTime));
    }
    
    // Install applications on mobile nodes
    for (uint32_t i = 0; i < mobileNodes.GetN(); i++)
    {
        Ptr<CompleteSystemApp> app = CreateObject<CompleteSystemApp>();
        app->Setup(9, true, 2.0, 0.7, 5);  // mobile, beacon every 2s, gossip enabled
        mobileNodes.Get(i)->AddApplication(app);
        app->SetStartTime(Seconds(1.0));
        app->SetStopTime(Seconds(simTime));
    }
    
    // Initiate gossip messages
    Ptr<CompleteSystemApp> gossipNode = DynamicCast<CompleteSystemApp>(
        staticNodes.Get(0)->GetApplication(0));
    Simulator::Schedule(Seconds(5.0), &CompleteSystemApp::InitiateGossip, 
                       gossipNode, "Important network update!");
    
    Ptr<CompleteSystemApp> gossipNode2 = DynamicCast<CompleteSystemApp>(
        mobileNodes.Get(0)->GetApplication(0));
    Simulator::Schedule(Seconds(15.0), &CompleteSystemApp::InitiateGossip, 
                       gossipNode2, "Mobile node discovery!");
    
    // Animation
    AnimationInterface anim("final-project.xml");
    
    // Color nodes
    for (uint32_t i = 0; i < staticNodes.GetN(); i++)
    {
        anim.UpdateNodeDescription(staticNodes.Get(i), "Base-" + std::to_string(i));
        anim.UpdateNodeColor(staticNodes.Get(i), 0, 0, 255);  // Blue
        anim.UpdateNodeSize(staticNodes.Get(i)->GetId(), 4.0, 4.0);
    }
    
    for (uint32_t i = 0; i < mobileNodes.GetN(); i++)
    {
        anim.UpdateNodeDescription(mobileNodes.Get(i), "Mobile-" + std::to_string(i));
        anim.UpdateNodeColor(mobileNodes.Get(i), 255, 0, 0);  // Red
        anim.UpdateNodeSize(mobileNodes.Get(i)->GetId(), 3.0, 3.0);
    }
    
    Simulator::Stop(Seconds(simTime));
    Simulator::Run();
    Simulator::Destroy();
    
    std::cout << "\n=== Simulation Complete ===" << std::endl;
    std::cout << "Animation file: final-project.xml" << std::endl;
    
    return 0;
}
```

- Ensure proper timing (apps start after nodes initialized)

- Enable logging: `LogComponentEnable("AppName", LOG_LEVEL_INFO);`
- Check application start/stop times
- Verify IP addresses assigned correctly
